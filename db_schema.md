``` SQL
create table
  public."Bookmark" (
    id bigint generated by default as identity,
    created_at timestamp with time zone not null default now(),
    user_id uuid not null default auth.uid (),
    post_id bigint not null,
    constraint Bookmark_pkey primary key (id),
    constraint public_Bookmark_post_id_fkey foreign key (post_id) references "Post" (id),
    constraint public_Bookmark_user_id_fkey foreign key (user_id) references "Profile" (id)
  ) tablespace pg_default;
```

```SQL
  create table
  public."Comment" (
    id bigint generated by default as identity,
    created_at timestamp with time zone not null default now(),
    user_id uuid null default auth.uid (),
    post_id bigint not null,
    content text null,
    likes json null,
    constraint Comment_pkey primary key (id),
    constraint public_Comment_post_id_fkey foreign key (post_id) references "Post" (id),
    constraint public_Comment_user_id_fkey foreign key (user_id) references "Profile" (id)
  ) tablespace pg_default;
```

```SQL
create table
  public."Post" (
    id bigint generated by default as identity,
    created_at timestamp with time zone not null default now(),
    user_id uuid null default auth.uid (),
    title character varying null,
    content text null,
    is_published boolean not null,
    is_deleted boolean not null,
    likes json null,
    tags json null,
    constraint Post_pkey primary key (id),
    constraint public_Post_user_id_fkey foreign key (user_id) references "Profile" (id)
  ) tablespace pg_default;
```

```SQL
create table
  public."Profile" (
    id uuid not null default auth.uid (),
    created_at timestamp with time zone not null default now(),
    username text not null,
    email character varying null,
    avatar_url text null,
    provider character varying null,
    tags json null,
    constraint Profile_pkey primary key (id),
    constraint Profile_username_key unique (username)
  ) tablespace pg_default;
```

create table

    constraint Reply_pkey primary key (id),
